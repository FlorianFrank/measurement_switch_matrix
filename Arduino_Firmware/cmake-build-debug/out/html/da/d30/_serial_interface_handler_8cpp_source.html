52/*static*/ Message SerialInterfaceHandler::ReadAndParseMessageBlocking(int *len)
53{
54 if(!len || *len == 0)
55 return Message(0, NOPE);
56
57 int pos = 0;
58 char tempData[MAX_COMMAND_LEN];
59 memset(tempData, 0x00, MAX_COMMAND_LEN);
60
61 // collect data until \n
62 int lenToRead = *len;
63 *len = 0;
64 do
65 {
66 *len += Serial.readBytes(&tempData[pos], lenToRead);
67 lenToRead =- *len;
68 pos += *len;
69 } while(!strstr(tempData, "\n") && lenToRead > 0);
70 Serial.write(tempData);
71
72
73 if(*len == 0)
74 return Message(0, NOPE);
75
76 return Message::CharArrayToMessage(tempData);
